jQuery Remote Clock

Basic usage

                    <div class="demo-clock-holder-1"></div>

                    <script>
                        $('.demo-clock-holder-1').remoteClock({
                            url: '/time'
                        });
                    </script>

Change sync interval

                    <div class="demo-clock-holder-2"></div>

                    <script>
                        $('.demo-clock-holder-2').remoteClock({
                            url: '/time',
                            syncInterval: 60000
                        });
                    </script>

Customize widget template

                    <div class="demo-clock-holder-3"></div>

                    <script>
                        $('.demo-clock-holder-3').remoteClock({
                            url: '/time',
                            template: '<i class="glyphicon glyphicon-time"></i>'
                                    + '<span class="time"></span>'
                        });
                    </script>

Use custom time parser

                    <div class="demo-clock-holder-4"></div>

                    <script>
                        $('.demo-clock-holder-4').remoteClock({
                            url: '/time',
                            parser: function (data) {
                                return Date.parse(data.time);
                            }
                        });
                    </script>
